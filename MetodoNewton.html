import numpy as np
import matplotlib.pyplot as plt


# rango para dibujar tangentes
x_plot = np.linspace(-12, 5, 400)


x_vals1 = []
x1 = -10

for i in range(10):
    x_vals1.append(x1)
    x1 = x1 - f(x1)/df(x1)


x_vals2 = []
x2 = -8

for i in range(10):
    x_vals2.append(x2)
    x2 = x2 - f(x2)/df(x2)



plt.figure(figsize=(18,6))

# función original
plt.plot(x_plot, f(x_plot), 'k', label="f(x)")

# -------- tangentes x0 = -10 --------
for x0 in x_vals1:
    y0 = f(x0)
    m = df(x0)
    tangent = y0 + m*(x_plot - x0)
    plt.plot(x_plot, tangent, '--', alpha=0.4, color='blue')

# -------- tangentes x0 = -8 --------
for x0 in x_vals2:
    y0 = f(x0)
    m = df(x0)
    tangent = y0 + m*(x_plot - x0)
    plt.plot(x_plot, tangent, '--', alpha=0.4, color='orange')

# puntos de iteración
plt.scatter(x_vals1, [f(x) for x in x_vals1], color='blue', label="x0 = -10")
plt.scatter(x_vals2, [f(x) for x in x_vals2], color='orange', label="x0 = -8")

plt.axhline(0, color='black', linewidth=1)
plt.title("Método de Newton con tangentes")
plt.legend()
plt.grid(True)

plt.show()
